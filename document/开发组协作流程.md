## 图二：模块数据流图（数据视角，显示谁产出什么文件、交给谁）

```mermaid
flowchart TB
    subgraph DAY12["第1-2天：基础准备"]
        direction LR
        AGREEMENT["三人约定<br/>标签:normal/risk<br/>CSV格式:5列<br/>image_path:相对路径"]
        A_D2["A 数据清洗<br/>→ dataset_v1.csv<br/>含 sample_id,text,<br/>image_path,label,split"]
    end

    subgraph DAY34["第3-4天：单模态并行"]
        direction LR
        A_TFIDF["A TF-IDF+逻辑回归<br/>→ text_tfidf_pred.csv"]
        B_RESNET["B ResNet18特征提取<br/>→ image_resnet_pred.csv"]
        A_BERT["A BERT文本向量<br/>→ text_bert_pred.csv<br/>+ 维度说明(如768维)"]
        C_EVAL3["C 评估函数<br/>→ 单模态指标表"]
        C_CHECK["C 检查<br/>sample_id一致性<br/>测试集一致性"]
    end

    subgraph DAY5["第5天：多模态融合 核心依赖日"]
        direction LR
        A_VEC["A 交付 文本向量<br/>+ 行为特征CSV<br/>⚠️ B的融合前置条件"]
        B_FUSION["B 拼接融合<br/>文本向量+图像向量+行为特征<br/>→ MLP分类器<br/>→ fusion_pred.csv"]
        C_EVAL5["C 评估融合结果<br/>→ 文本 vs 图像 vs 融合<br/>对比指标表"]
    end

    subgraph DAY6["第6天：消融实验"]
        direction LR
        ABLATION["5组消融对比<br/>① 仅文本 ② 仅图像 ③ 仅行为<br/>④ 文本+图像 ⑤ 文本+图像+行为"]
        C_CHARTS["C 画图<br/>→ 混淆矩阵 ROC/PR曲线<br/>→ 消融柱状图 错误案例"]
        A_ERR["A 文本误判案例"]
        B_ERR["B 图像误判案例"]
    end

    subgraph DAY7["第7天：代码冻结交接"]
        direction LR
        FREEZE["代码冻结<br/>git tag v1.0-freeze"]
        HANDOFF["C 打包交接包<br/>→ 数据说明 模型说明<br/>→ 最终指标表 图表清单<br/>→ 错误案例 复现命令<br/>→ 技术解释"]
        DE["D/E 报告答辩组<br/>接收交接包"]
    end

    AGREEMENT --> A_D2
    A_D2 -->|CSV含image_path| B_RESNET
    A_D2 -->|同一份数据| A_TFIDF
    A_TFIDF -->|预测CSV| C_EVAL3
    B_RESNET -->|预测CSV| C_EVAL3
    C_EVAL3 --> C_CHECK
    A_BERT -->|文本向量+维度| A_VEC
    A_VEC -->|文本向量CSV 行为特征CSV| B_FUSION
    B_RESNET -->|图像向量| B_FUSION
    B_FUSION -->|融合预测CSV| C_EVAL5
    C_CHECK --> C_EVAL5
    C_EVAL5 -->|单模态+融合对比表| ABLATION
    ABLATION --> C_CHARTS
    A_ERR --> C_CHARTS
    B_ERR --> C_CHARTS
    C_CHARTS --> FREEZE
    FREEZE --> HANDOFF
    HANDOFF --> DE
```
